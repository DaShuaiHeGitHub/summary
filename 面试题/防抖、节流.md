# 防抖
```javascript
// 防抖：触发函数 N秒 后才执行，如果再次触发则时间从新计算
function debounce(fn, wait = 500) {
    let timer = null
    return function(...args) {
    timer && clearTimeout(timer)

    timer = setTimeout(() => {
        fn.apply(this, args)
    }, wait);
    }
}
```

# 节流

```javascript
// 节流：N秒 内只触发一次
function throttle(fn, delay = 500) {
    let prevDate = 0

    return function(...args) {
    const now = Date.now()
    if(now - prevDate > delay) {
        prevDate = now
        fn.apply(this, args)
    }
    }
}
```
